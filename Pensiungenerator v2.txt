/** REF: toUpper() */
function pensiunGen_upper_(str) {
  return str ? String(str).toUpperCase().trim() : '';
}

/** REF: formatTanggalIndonesia() -- versi UPPERCASE, dipakai seluruh tanggal DPCP. */
function pensiunGen_tanggalUpper_(value) {
  const formatted = formatTanggalIndonesia_(value);
  return formatted ? formatted.toUpperCase() : '';
}

/** REF: formatTanggalIndonesiaBiasa() -- versi BUKAN uppercase, khusus {{tanggal_super}}. */
function pensiunGen_tanggalBiasa_(value) {
  return formatTanggalIndonesia_(value) || '';
}

/** REF: dash() -- "-" bila kosong. */
function pensiunGen_dash_(val) {
  const s = (val !== null && val !== undefined) ? String(val).trim() : '';
  return s === '' ? '-' : s;
}

/** REF: toUpperOrDash() */
function pensiunGen_upperOrDash_(val) {
  const s = pensiunGen_upper_(val);
  return s === '' ? '-' : s;
}

/** REF: formatTanggalOrDash() */
function pensiunGen_tanggalOrDash_(val) {
  if (!val || String(val).trim() === '') return '-';
  const r = pensiunGen_tanggalUpper_(val);
  return r === '' ? '-' : r;
}

/**
 * REF: buildFamilyRows() -- susun array baris keluarga siap-cetak:
 * - baris yang seluruh field-nya kosong DIBUANG (tidak dicetak jadi baris)
 * - baris yang terisi sebagian -> field kosong otomatis "-"
 * - kalau TIDAK ADA data sama sekali -> 1 baris dash ("-") semua field
 * - kolom "no" otomatis berurutan 1,2,3,... sesuai baris yang tercetak
 */
function pensiunGen_buildFamilyRows_(rawArray, fieldNames, dateFields, upperFields) {
  const filtered = (rawArray || []).filter(item => {
    if (!item) return false;
    return fieldNames.some(f => item[f] !== null && item[f] !== undefined && String(item[f]).trim() !== '');
  });

  if (filtered.length === 0) {
    const dashRow = { no: '-' };
    fieldNames.forEach(f => { dashRow[f] = '-'; });
    return [dashRow];
  }

  return filtered.map((item, idx) => {
    const row = { no: String(idx + 1) };
    fieldNames.forEach(f => {
      const val = item[f];
      if (dateFields.indexOf(f) !== -1) row[f] = pensiunGen_tanggalOrDash_(val);
      else if (upperFields.indexOf(f) !== -1) row[f] = pensiunGen_upperOrDash_(val);
      else row[f] = pensiunGen_dash_(val);
    });
    return row;
  });
}

/**
 * REF: bagian "Variabel jenis pensiun" + "{{akhir_tmt_kerja}}" di dalam
 * generateDPCPDocument(). Dipisah jadi fungsi sendiri di sini supaya
 * bisa diuji terpisah.
 */
function pensiunGen_hitungJenisPensiunFields_(formData) {
  let jp1 = '', jp2 = '', jp3 = '', namaTtd = '', nipTtd = '';

  const jp = String(formData.jenis_pensiun || '').trim();

  if (jp === 'BUP') {
    jp1 = 'BUP';
    jp2 = 'MENCAPAI BATAS USIA PENSIUN';
    jp3 = 'PEGAWAI NEGERI SIPIL YANG BERSANGKUTAN';
    namaTtd = pensiunGen_upper_(formData.nama_lengkap);
    nipTtd = formData.nip ? 'NIP. ' + formData.nip : '';
  } else if (jp === 'Meninggal') {
    jp1 = 'MENINGGAL';
    jp2 = 'AKAN DIBERHENTIKAN/YANG MENINGGAL DUNIA, TEWAS, ATAU HILANG';
    jp3 = 'SUAMI/ISTRI PNS YANG BERSANGKUTAN';
    const pasangan1 = (formData.pasangan && formData.pasangan[0]) || {};
    const namaPas = formData.nama_suami_istri_1 || pasangan1.nama_pasangan || formData.nama_pasangan || '';
    namaTtd = pensiunGen_upper_(namaPas);
    const nikPas = formData.nik_suami_istri_1 || pasangan1.nik || formData.nik_pasangan || '';
    nipTtd = nikPas ? 'NIK. ' + nikPas : (formData.nip ? 'NIP. ' + formData.nip : '');
  } else if (jp === 'Uzur') {
    jp1 = 'UZUR';
    jp2 = 'TIDAK CAKAP JASMANI DAN/ATAU ROHANI';
    jp3 = 'PEGAWAI NEGERI SIPIL YANG BERSANGKUTAN';
    namaTtd = pensiunGen_upper_(formData.nama_lengkap);
    nipTtd = formData.nip ? 'NIP. ' + formData.nip : '';
  } else if (jp === 'Diberhentikan') {
    jp1 = 'DIBERHENTIKAN';
    jp2 = 'AKAN DIBERHENTIKAN/YANG MENINGGAL DUNIA, TEWAS, ATAU HILANG';
    jp3 = 'PEGAWAI NEGERI SIPIL YANG BERSANGKUTAN';
    namaTtd = pensiunGen_upper_(formData.nama_lengkap);
    nipTtd = formData.nip ? 'NIP. ' + formData.nip : '';
  } else if (jp === 'Pengunduran Diri') {
    jp1 = 'PENGUNDURAN DIRI';
    jp2 = 'MENGAJUKAN PERMINTAAN SENDIRI';
    jp3 = 'PEGAWAI NEGERI SIPIL YANG BERSANGKUTAN';
    namaTtd = pensiunGen_upper_(formData.nama_lengkap);
    nipTtd = formData.nip ? 'NIP. ' + formData.nip : '';
  }

  let akhirTmtKerja;
  if (jp === 'BUP') {
    akhirTmtKerja = pensiunGen_tanggalUpper_(formData.tmt_pensiun);
  } else {
    akhirTmtKerja = pensiunGen_tanggalUpper_(formData.tgl_peristiwa_asli || formData.bulan_terakhir_bekerja || formData.tmt_pensiun);
  }

  return { jp1, jp2, jp3, namaTtd, nipTtd, akhirTmtKerja };
}

function pensiunGen_formatGajiRupiah_(val) {
  if (!val) return '-';
  const numStr = String(val).replace(/[^0-9]/g, '');
  if (!numStr) return String(val);
  const num = parseInt(numStr, 10);
  return 'Rp ' + num.toLocaleString('id-ID') + ',-';
}

function makeAllStringsUpper_(obj) {
  if (obj === null || obj === undefined) return obj;
  if (typeof obj === 'string') return obj.toUpperCase();
  if (Array.isArray(obj)) return obj.map(makeAllStringsUpper_);
  if (typeof obj === 'object') {
    const res = {};
    for (const key of Object.keys(obj)) {
      if (key === '_fotoBase64') res[key] = obj[key];
      else res[key] = makeAllStringsUpper_(obj[key]);
    }
    return res;
  }
  return obj;
}

    nama: pensiunGen_upper_(formData.nama),

    tmt_pengangkat: pensiunGen_tanggalUpper_(formData.tmt_cpns),
    golongan: pensiunGen_upper_(formData.golongan),
    pangkat: pangkatGabungan,
    tmt_gol: fmtTmtGol,
    jabatan: pensiunGen_upper_(formData.jabatan),
    tmt_jab: pensiunGen_tanggalUpper_(formData.tmt_jab),
    tmt_pensiun: pensiunGen_tanggalUpper_(formData.tmt_pensiun),
    unit_es_ii: pensiunGen_upper_(formData.unit_es_ii),
    unit_kerja: pensiunGen_upper_(formData.unit_es_ii),
    gaji_pokok: pensiunGen_formatGajiRupiah_(formData.gaji_pokok),
    mk_kp_terakhir: pensiunGen_upper_(formData.mk_kp_terakhir) || '0 TAHUN 0 BULAN',
    mk_golongan: pensiunGen_upper_(formData.mk_golongan),
    mk_pns: pensiunGen_upper_(formData.mk_pns),
    mk_pensiun: pensiunGen_upper_(formData.mk_pensiun),
    alamat: pensiunGen_upper_(formData.alamat),
    pendidikan_pertama: pensiunGen_upper_(formData.pendidikan_pertama),
    cltn: pensiunGen_upper_(formData.cltn) || '0 TAHUN 0 BULAN',
    pmk: pensiunGen_upper_(formData.pmk) || '0 TAHUN 0 BULAN',
    akhir_tmt_kerja: jenisPensiunFields.akhirTmtKerja,
    bulan_terakhir_bekerja: pensiunGen_tanggalUpper_(formData.bulan_terakhir_bekerja),

    anak: pensiunGen_buildFamilyRows_(formData.anak, PENSIUN_ANAK_FIELDS, PENSIUN_ANAK_DATE_FIELDS, PENSIUN_ANAK_FIELDS)
  };

  ctx._fotoBase64 = formData.fotoBase64 || '';
  return makeAllStringsUpper_(ctx);
}

/**
 * REF: bagian "PROSES FOTO" di generateDPCPDocument(). Dipanggil sebagai
 * preProcessFn oleh renderTemplateCopy_ SEBELUM tag {{..}} lain diisi,
 * supaya literal "{{foto}}" masih ada untuk dicari lewat findText() --
 * kalau dijalankan SESUDAH pass tag umum, tag itu sudah keburu diganti
 * string kosong (foto dianggap identifier biasa yang tidak dikenal).
 */
function pensiunGen_insertFoto_(doc, dataContext) {
  const body = doc.getBody();
  const fotoBase64 = dataContext._fotoBase64;

  if (!fotoBase64) {
    try {
      body.replaceText('\\{\\{\\s*foto\\s*\\}\\}', ' ');
    } catch (e) {
      Logger.log('Foto clear err: ' + e.message);
    }
    return;
  }

  const found = body.findText('{{foto}}') || body.findText('{{ foto }}');
  if (!found) return; // template tidak punya placeholder foto -- lewati diam-diam

  const textObj = found.getElement();
  const textString = textObj.getText();
  const parentParagraph = textObj.getParent().asParagraph();

  try {
    const splitData = fotoBase64.split(',');
    const contentType = splitData[0].match(/:(.*?);/)[1];
    const rawBase64 = splitData[1];
    const imageBlob = Utilities.newBlob(Utilities.base64Decode(rawBase64), contentType, 'foto_pegawai.jpg');

    const inlineImage = parentParagraph.appendInlineImage(imageBlob);
    inlineImage.setWidth(120);
    inlineImage.setHeight(160);

    const remText = textString.replace('{{foto}}', '').replace('{{ foto }}', '');
    if (remText.trim() === '') {
      textObj.setText(' ');
    } else {
      textObj.setText(remText);
    }
  } catch (imgErr) {
    const fallbackText = textString.replace('{{foto}}', '[Foto]').replace('{{ foto }}', '[Foto]');
    if (fallbackText.trim() === '') {
      textObj.setText(' ');
    } else {
      textObj.setText(fallbackText);
    }
  }
}

function pensiunGen_ensureSaveSheet_() {
  const ss = getSpreadsheet_();
  let sheet = ss.getSheetByName(PENSIUN_SAVE_SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(PENSIUN_SAVE_SHEET_NAME);
    sheet.appendRow(PENSIUN_SAVE_HEADERS);
    sheet.setFrozenRows(1);
  }
  return sheet;
}

/** REF: generateAnakRows() -- 5 slot anak tetap (dash bila kosong), khusus untuk baris log spreadsheet. */
function pensiunGen_anakLogValues_(anakArray) {
  const rows = [];
  const arr = anakArray || [];
  for (let i = 0; i < 5; i++) {
    const item = arr[i] || {};
    const useDash = PENSIUN_ANAK_FIELDS.some(f => item[f] && String(item[f]).trim() !== '');
    const d = v => (useDash ? pensiunGen_dash_(v) : (v || ''));
    const du = v => (useDash ? pensiunGen_upperOrDash_(v) : pensiunGen_upper_(v));
    rows.push(d(item.nik), du(item.nama_anak), d(item.tg_lahir_anak), du(item.nama_ortu), d(item.keterangan_anak));
  }
  return rows;
}

    pensiunGen_upper_(formData.nama),
    pensiunGen_upper_(formData.tmp_lhr),
    formData.tgl_lhr || '',
    formData.tmt_cpns || '',

    pensiunGen_upper_(formData.golongan),
    formData.tmt_gol || '',
    pensiunGen_upper_(formData.jabatan),
    formData.tmt_jab || '',
    formData.tmt_pensiun || '',
    formData.bulan_terakhir_bekerja || '',
    pensiunGen_upper_(formData.unit_es_ii),
    formData.gaji_pokok || '',
    formData.mk_kp_terakhir || '',
    formData.mk_sebelum_cpns || '',
    formData.cltn || '',
    formData.pmk || '',

  sheet.appendRow(row);
}

/**
 * REF: getLatestSavedGenerateData() -- ambil histori generate TERAKHIR
 * seorang pegawai dari sheet "Save data" untuk fitur "gunakan data
 * generate sebelumnya" di form DPCP. Return null bila belum pernah ada.
 */
function pensiunGen_getLatestSavedData_(nip) {
  if (!nip) return null;
  const ss = getSpreadsheet_();
  const sheet = ss.getSheetByName(PENSIUN_SAVE_SHEET_NAME);
  if (!sheet) return null;

  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return null;
  const headers = data[0];
  const idxNip = headers.indexOf('NIP');
  if (idxNip === -1) return null;

  let lastRow = null;
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][idxNip] || '').trim() === String(nip).trim()) lastRow = data[i];
  }
  if (!lastRow) return null;

  function col(headerName) {
    const idx = headers.indexOf(headerName);
    if (idx === -1) return '';
    const val = lastRow[idx];
    if (val instanceof Date) return formatDateYMD_(val);
    return (val === null || val === undefined) ? '' : String(val);
  }
  function cleanVal(v) { return (v === '-') ? '' : v; }

  function pasanganFromCols(suffix) {
    const nik = cleanVal(col(`NIK Pasangan ${suffix}`));
    const nama = cleanVal(col(`Nama Pasangan ${suffix}`));
    const tglLahir = cleanVal(col(`Tgl Lahir Pasangan ${suffix}`));
    const tglNikah = cleanVal(col(`Tgl Nikah Pasangan ${suffix}`));
    const tglCerai = cleanVal(col(`Tgl Cerai/MD Pasangan ${suffix}`));
    const ket = cleanVal(col(`Istri Ke-${suffix}`));
    const hasValue = [nik, nama, tglLahir, tglNikah, tglCerai, ket].some(v => v && String(v).trim() !== '');
    if (!hasValue) return null;
    return { nik, nama_pasangan: nama, tgl_lahir_pasangan: tglLahir, tgl_kawin: tglNikah, tgl_cerai: tglCerai, keterangan_pasangan: ket };
  }
  function anakFromCols(suffix) {
    const nik = cleanVal(col(`NIK Anak ${suffix}`));
    const nama = cleanVal(col(`Nama Anak ${suffix}`));
    const tglLahir = cleanVal(col(`Tgl Lahir Anak ${suffix}`));
    const namaOrtu = cleanVal(col(`Nama Ortu Anak ${suffix}`));
    const ket = cleanVal(col(`Ket Anak ${suffix}`));
    const hasValue = [nik, nama, tglLahir, namaOrtu, ket].some(v => v && String(v).trim() !== '');
    if (!hasValue) return null;
    return { nik, nama_anak: nama, tg_lahir_anak: tglLahir, nama_ortu: namaOrtu, keterangan_anak: ket };
  }

  return {
    jenis_pensiun: col('Jenis Pensiun'),
    nip: col('NIP'),
    nama_lengkap: col('Nama Lengkap'),
    nama: col('Nama Panggilan'),
    tmp_lhr: col('Tempat Lahir'),
    tgl_lhr: col('Tanggal Lahir'),
    tmt_cpns: col('TMT CPNS'),
    pangkat: col('Pangkat'),
    golongan: col('Golongan'),
    tmt_gol: col('TMT Golongan'),
    jabatan: col('Jabatan'),
    tmt_jab: col('TMT Jabatan'),
    tmt_pensiun: col('TMT Pensiun'),
    bulan_terakhir_bekerja: col('Bulan Terakhir Bekerja'),
    unit_es_ii: col('Unit Kerja'),
    gaji_pokok: col('Gaji Pokok'),
    mk_kp_terakhir: col('MK KP Terakhir'),
    mk_sebelum_cpns: col('MK Sebelum CPNS'),
    cltn: col('CLTN'),
    pmk: col('PMK'),
    pendidikan_pertama: col('Pendidikan Pertama'),
    alamat: col('Alamat'),
    pasangan: [1, 2].map(pasanganFromCols).filter(v => !!v),
    anak: [1, 2, 3, 4, 5].map(anakFromCols).filter(v => !!v)
  };
}

/** Dipanggil dari client untuk fitur "gunakan data generate sebelumnya". */

/**
 * Pindahkan file Drive ke folder tujuan (REF: output DPCP/SUPER tetap
 * Google Docs, disimpan di TARGET_FOLDER_ID -- bukan diexport PDF seperti
 * Kenaikan Pangkat). Pakai cara klasik add/removeFile supaya kompatibel
 * di semua versi Apps Script (tidak bergantung pada File.moveTo()).
 */
/** REVISI: dipindahkan ke Utils.gs sebagai pindahFileKeFolder_ (dipakai bersama KP/Pensiun/Kontrak). */

/** PREVIEW DPCP -- embed sementara, TIDAK disimpan permanen, TIDAK ditulis ke "Save data". */
function previewDpcpDocument(sessionToken, formData) {
  requireRole_(sessionToken, ['admin', 'super_admin']);
  try {
    const templates = getTemplatesForService(sessionToken, 'Pensiun', 'DPCP');
    if (!templates.length) return { success: false, message: 'Template DPCP belum terdaftar. Tambahkan di menu Template.' };

    const dataContext = pensiunGen_buildDpcpContext_(formData);
    const tempName = '[PREVIEW] DPCP - ' + sanitizeFileName_(dataContext.nama_lengkap || 'pegawai') + ' - ' + new Date().getTime();

    const copyFile = renderTemplateCopy_(templates[0].fileId, dataContext, tempName, pensiunGen_insertFoto_);
    try { copyFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch(e) { Logger.log('Share err: '+e.message); }

    return {
      success: true,
      previewUrl: `https://docs.google.com/document/d/${copyFile.getId()}/preview`,
      tempFileId: copyFile.getId()
    };
  } catch (err) {
    return { success: false, message: 'Gagal membuat preview DPCP: ' + err.message };
  }
}

/**
 * GENERATE DPCP -- REF: output akhir TETAP Google Docs (bukan PDF),
 * disimpan ke TARGET_FOLDER_ID (di sini: CONFIG.OUTPUT_FOLDER_ID, folder
 * output yang sama dipakai seluruh sistem). Jika formData.isSaveActive,
 * turut menulis baris log ke sheet "Save data" (REF: bagian SIMPAN KE
 * SPREADSHEET).
 */
function generateDpcpDocument(sessionToken, formData) {
  const pemroses = requireRole_(sessionToken, ['admin', 'super_admin']);
  try {
    const templates = getTemplatesForService(sessionToken, 'Pensiun', 'DPCP');
    if (!templates.length) return { success: false, message: 'Template DPCP belum terdaftar. Tambahkan di menu Template.' };

    const dataContext = pensiunGen_buildDpcpContext_(formData);
    const namaUtama = dataContext.nama_lengkap || 'PEGAWAI';
    const fileName = sanitizeFileName_(`DPCP_${namaUtama}`);

    const docCopy = renderTemplateCopy_(templates[0].fileId, dataContext, fileName, pensiunGen_insertFoto_);

    // REVISI: folder khusus "Pensiun" (terpisah dari Kenaikan Pangkat/Kontrak).
    const outputFolder = DriveApp.getFolderById(CONFIG.FOLDER_PENSIUN_OUTPUT);
    pindahFileKeFolder_(docCopy, outputFolder);
    try { docCopy.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch(e) { Logger.log('Share err: '+e.message); }

    if (formData.isSaveActive) {
      pensiunGen_saveLog_(formData);
    }

    // Kalau NIP ini berasal dari usulan Pensiun yang masih "Diajukan",
    // tandai DPCP selesai -- status baru berubah ke notif akhir begitu
    // SUPER juga selesai (lihat PensiunUsulanService.gs).
    if (formData.nip) tandaiDokumenPensiunSelesai_(formData.nip, 'dpcp', pemroses.nip);

    return {
      success: true,
      message: 'Dokumen DPCP berhasil dibuat' + (formData.isSaveActive ? ' & record database berhasil dicatat.' : '.'),
      docUrl: docCopy.getUrl(),
      viewUrl: docCopy.getUrl()
    };
  } catch (err) {
    return { success: false, message: 'Gagal generate dokumen DPCP: ' + err.message };
  }
}

/**
 * Bangun dataContext untuk dokumen SUPER. REF: CATATAN mapping {{pangkat}}
 * -- template SUPER tidak punya placeholder {{golongan}} terpisah, jadi
 * dipakai format gabungan "PANGKAT/GOLONGAN/TMT GOLONGAN" sama seperti
 * konvensi {{pangkat}} di DPCP. Beda dari DPCP, nilai SUPER TIDAK
 * di-uppercase -- ditulis persis sesuai isian sistem.
 */
function pensiunGen_buildSuperContext_(formData) {
  formData = formData || {};
  const fmtTmtGol = pensiunGen_tanggalBiasa_(formData.tmt_gol);
  const pangkatGabungan = `${formData.pangkat || ''}/${formData.golongan || ''}/${fmtTmtGol}`;

  return {
    nomor_super_tidak_hukdis_sedangberat: formData.nomor_super_tidak_hukdis_sedangberat || '',
    nama_lengkap: formData.nama_lengkap || '',
    nip: formData.nip || '',
    pangkat: pangkatGabungan,
    jabatan: formData.jabatan || '',
    tanggal_super: pensiunGen_tanggalBiasa_(new Date()),
    nomor_super_pidana: formData.nomor_super_pidana || ''
  };
}
