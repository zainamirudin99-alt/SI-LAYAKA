/**
 * =================================================================
 * DOCUMENTGENERATOR.GS
 * Menggabungkan data pegawai + data form ke dalam template Google
 * Docs, lalu menghasilkan PREVIEW (embed) atau file PDF FINAL
 * tersimpan otomatis ke folder Drive output.
 *
 * PEMBARUAN (fitur "copy/tambah data" Opsi A & B):
 * previewDocument() dan generateDocument() sekarang menerima payload
 * berbentuk BANYAK entri sekaligus (payload.entries = array), sehingga
 * beberapa isian Opsi A/B (mis. 2 pegawai atau 2 periode berbeda) bisa
 * digabung jadi SATU dokumen/PDF akhir (satu entri = satu "halaman"
 * hasil render template, digabung berurutan dengan page break).
 * Payload lama (formData + targetNip tunggal) TETAP didukung untuk
 * kompatibilitas ke belakang — otomatis diperlakukan sebagai 1 entri.
 * =================================================================
 */

/**
 * Membangun dataContext gabungan: data pegawai (dari sheet) + data form
 * (dari input pengguna) + field turunan khusus DocGen (lihat DocGen.gs).
 *
 * Field turunan (mis. total_ak_baru, rekomendasi) DIHITUNG oleh
 * docGen_buildDerivedFields_ di DocGen.gs sebagai NILAI FALLBACK — kalau
 * template Google Docs yang dipakai punya rumus "set" sendiri untuk field
 * yang sama, hasil rumus template itulah yang dipakai (diproses TemplateEngine
 * setelah dataContext ini disiapkan). TIDAK memakai/mengubah mesin di
 * EligibilityService.gs. Angka pada dokumen resmi hasil generate karenanya
 * bisa berbeda dari angka estimasi di halaman Check Eligible; ini memang
 * disengaja (lihat catatan di DocGen.gs).
 *
 * ALIAS NAMA FIELD: form Opsi B (JavaScriptClient.html) memakai nama field
 * pendek untuk sebagian input (mis. "jumlah_angka_kredit_diperoleh",
 * "ada_ijazah_baru_2023"), sedangkan tag di dalam template Google Docs
 * memakai nama panjang ("jumlah_angka_kredit_yang_diperoleh_saat_integrasi",
 * "ada_ijazah_baru_setelah_2023"). Tanpa alias ini, rumus "set" di dalam
 * template membaca nilai undefined/NaN meski form sudah diisi lengkap.
 * Kedua nama sengaja disediakan sekaligus di sini supaya template BOLEH
 * memakai nama pendek ATAU panjang, dan form tidak perlu diubah.
 *
 * @param {Object} employee
 * @param {Object} formData
 * @param {string} [subLayanan]  'AK Konversi Tahunan' | 'AK Konversi Kumulatif' | lainnya
 */
function buildDataContext_(employee, formData, subLayanan) {
  formData = formData || {};
  const derived = docGen_buildDerivedFields_(employee, formData, subLayanan);

  const alias = {};
  if (formData.jumlah_angka_kredit_diperoleh !== undefined) {
    alias.jumlah_angka_kredit_yang_diperoleh_saat_integrasi = formData.jumlah_angka_kredit_diperoleh;
  }
  if (formData.ada_ijazah_baru_2023 !== undefined) {
    alias.ada_ijazah_baru_setelah_2023 = formData.ada_ijazah_baru_2023;
  }

  // Pemetaan alias untuk nama pendek/panjang (GDocs)
  const map = {
    tgl_lhr: 'tanggal_lahir',
    tmp_lhr: 'tempat_lahir',
    jns_kel: 'jenis_kelamin',
    tmt_gol: 'tmt_golongan',
    tmt_jab: 'tmt_jabatan',
    unit_es_ii: 'unit_kerja',
    karpeg: 'kartu_pegawai'
  };
  for (const key in map) {
    const val = map[key];
    if (employee && employee[key] !== undefined && alias[val] === undefined) alias[val] = employee[key];
    if (employee && employee[val] !== undefined && alias[key] === undefined) alias[key] = employee[val];
    if (formData && formData[key] !== undefined && alias[val] === undefined) alias[val] = formData[key];
    if (formData && formData[val] !== undefined && alias[key] === undefined) alias[key] = formData[val];
  }

  const ctx = Object.assign({}, employee, formData, alias, derived);
  // "today" tersedia otomatis lewat resolveIdentifier_, tidak perlu disuntik di sini.
  return ctx;
}

/**
 * Salin template Google Docs berdasarkan file ID, isi seluruh tag,
 * lalu kembalikan objek Document yang sudah ter-render (belum di-trash).
 * @param {string} templateFileId
 * @param {Object} dataContext
 * @param {string} tempName
 * @param {Function} [preProcessFn]  Opsional: (doc, dataContext) => void,
 *   dijalankan SEBELUM tag {{..}} diisi (dipakai mis. oleh modul Pensiun
 *   untuk menyisipkan foto ke {{foto}} sebelum tag itu "dihabiskan" oleh
 *   pass evaluasi umum). Tidak dipakai oleh Kenaikan Pangkat -- default
 *   tanpa efek apa pun bila tidak diisi.
 */
function renderTemplateCopy_(templateFileId, dataContext, tempName, preProcessFn) {
  const templateFile = DriveApp.getFileById(templateFileId);
  const copy = templateFile.makeCopy(tempName);
  const doc = DocumentApp.openById(copy.getId());
  const body = doc.getBody();

  if (preProcessFn) preProcessFn(doc, dataContext);

  renderDocumentBody_(body, dataContext);
  // Proses juga header & footer bila ada tag di sana
  const header = doc.getHeader();
  if (header) renderDocumentBody_(header, dataContext);
  const footer = doc.getFooter();
  if (footer) renderDocumentBody_(footer, dataContext);

  doc.saveAndClose();
  return copy; // File Drive
}

/**
 * Susun payload menjadi daftar entri siap-render: masing-masing berisi
 * {employee, dataContext}. Menerima BAIK payload.entries (array — fitur
 * "copy/tambah data") MAUPUN payload.formData + payload.targetNip tunggal
 * (kompatibel dengan pemanggilan versi lama).
 * @param {string} sessionToken
 * @param {Object} employeeSession  Data pegawai pemilik sesi (fallback bila targetNip kosong)
 * @param {Object} payload          { templateFileId, subLayanan, entries? , formData?, targetNip? }
 */
function buildEntriesContexts_(sessionToken, employeeSession, payload) {
  const rawEntries = (Array.isArray(payload.entries) && payload.entries.length)
    ? payload.entries
    : [{ formData: payload.formData, targetNip: payload.targetNip }];

  return rawEntries
    .filter(e => e && e.formData)
    .map(e => {
      const employee = e.targetNip
        ? getEmployeeFullData(sessionToken, e.targetNip)
        : employeeSession;
      return {
        employee: employee,
        dataContext: buildDataContext_(employee, e.formData || {}, payload.subLayanan)
      };
    });
}

/**
 * Gabungkan hasil render SATU template untuk beberapa entri menjadi SATU
 * dokumen Google Docs (dipisah page break per entri), lalu kembalikan file
 * Drive-nya (BELUM dikonversi PDF — dipakai baik untuk preview maupun
 * generate final). Header/footer dokumen gabungan mengikuti entri PERTAMA
 * saja (konvensi surat yang dicetak berkelanjutan dalam satu file).
 */
function renderMultiEntryDocument_(templateFileId, entries, tempName) {
  const templateFile = DriveApp.getFileById(templateFileId);

  // Entri pertama = dokumen utama yang menampung SEMUA entri.
  const mainCopy = templateFile.makeCopy(tempName);
  const mainDoc = DocumentApp.openById(mainCopy.getId());
  renderDocumentBody_(mainDoc.getBody(), entries[0].dataContext);
  const mainHeader = mainDoc.getHeader();
  if (mainHeader) renderDocumentBody_(mainHeader, entries[0].dataContext);
  const mainFooter = mainDoc.getFooter();
  if (mainFooter) renderDocumentBody_(mainFooter, entries[0].dataContext);

  // Entri ke-2 dst: render template SEKALI LAGI di salinan sementara
  // terpisah (supaya variabel "set" antar-entri tidak saling tertukar),
  // lalu gabungkan hasilnya (dengan page break) ke dokumen utama, baru
  // salinan sementara tsb dibuang.
  for (let i = 1; i < entries.length; i++) {
    const extraCopy = templateFile.makeCopy(tempName + ' (bagian ' + (i + 1) + ')');
    const extraDoc = DocumentApp.openById(extraCopy.getId());
    renderDocumentBody_(extraDoc.getBody(), entries[i].dataContext);

    mainDoc.getBody().appendPageBreak();
    appendBodyContent_(mainDoc.getBody(), extraDoc.getBody());

    extraDoc.saveAndClose();
    extraCopy.setTrashed(true);
  }

  mainDoc.saveAndClose();
  return mainCopy;
}

/**
 * Salin seluruh isi sourceBody ke akhir targetBody, elemen demi elemen,
 * mempertahankan jenis elemen (paragraf, tabel, daftar, dsb). Dipakai
 * untuk menggabungkan beberapa hasil render template (fitur multi-entri
 * Opsi A/B) menjadi SATU dokumen/file akhir.
 */
function appendBodyContent_(targetBody, sourceBody) {
  const total = sourceBody.getNumChildren();
  for (let i = 0; i < total; i++) {
    appendElementByType_(targetBody, sourceBody.getChild(i));
  }
}

function appendElementByType_(targetBody, el) {
  const type = el.getType();
  switch (type) {
    case DocumentApp.ElementType.PARAGRAPH:
      targetBody.appendParagraph(el.copy());
      break;
    case DocumentApp.ElementType.TABLE:
      targetBody.appendTable(el.copy());
      break;
    case DocumentApp.ElementType.LIST_ITEM:
      targetBody.appendListItem(el.copy());
      break;
    case DocumentApp.ElementType.INLINE_IMAGE:
      targetBody.appendImage(el.copy());
      break;
    case DocumentApp.ElementType.HORIZONTAL_RULE:
      targetBody.appendHorizontalRule();
      break;
    case DocumentApp.ElementType.PAGE_BREAK:
      targetBody.appendPageBreak();
      break;
    default:
      // Tipe elemen lain (jarang dipakai di template surat) — dilewati
      // dengan aman agar proses gabung tidak gagal total.
      break;
  }
}

/**
 * PREVIEW: menghasilkan salinan sementara (bisa berisi >1 entri digabung)
 * dan mengembalikan URL embed (iframe) untuk ditampilkan di frontend.
 */
function previewDocument(sessionToken, payload) {
  const employeeSession = requireRole_(sessionToken, ['admin', 'super_admin']);
  try {
    if (!payload.templateFileId) return { success: false, message: 'Template belum dipilih.' };

    const entries = buildEntriesContexts_(sessionToken, employeeSession, payload);
    if (!entries.length) return { success: false, message: 'Data isian kosong — minimal 1 entri diperlukan.' };

    const namaAcuan = entries[0].employee.nama_lengkap || entries[0].employee.nama || 'pegawai';
    const tempName = '[PREVIEW] ' + sanitizeFileName_(namaAcuan) +
      (entries.length > 1 ? ` (+${entries.length - 1} lainnya)` : '') + ' - ' + new Date().getTime();

    const copyFile = renderMultiEntryDocument_(payload.templateFileId, entries, tempName);

    // Otomatis buka akses "siapapun yang punya link bisa melihat" agar
    // iframe preview langsung tampil tanpa dialog "Request Access".
    try {
      copyFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    } catch (shareErr) {
      Logger.log('Gagal set sharing preview: ' + shareErr.message);
      // Tetap lanjut — preview mungkin masih bisa dibuka jika user sudah login Google.
    }

    return {
      success: true,
      previewUrl: `https://docs.google.com/document/d/${copyFile.getId()}/preview`,
      tempFileId: copyFile.getId(),
      jumlahEntri: entries.length
    };
  } catch (err) {
    return { success: false, message: 'Gagal membuat preview: ' + err.message };
  }
}


/**
 * GENERATE: render dokumen final (bisa berisi >1 entri digabung dalam
 * satu file), konversi ke PDF, simpan otomatis ke folder Drive output,
 * hapus file Docs sementara, kembalikan link download langsung ke frontend.
 */
function generateDocument(sessionToken, payload) {
  const employeeSession = requireRole_(sessionToken, ['admin', 'super_admin']);
  try {
    if (!payload.templateFileId) return { success: false, message: 'Template belum dipilih.' };

    const entries = buildEntriesContexts_(sessionToken, employeeSession, payload);
    if (!entries.length) return { success: false, message: 'Data isian kosong — minimal 1 entri diperlukan.' };

    const namaUtama = entries[0].employee.nama_lengkap || entries[0].employee.nama || 'pegawai';
    const fileNameBase = entries.length > 1
      ? `${namaUtama}-dkk-${entries.length}data-${payload.subLayanan}`
      : `${namaUtama}-${entries[0].employee.nip}-${payload.subLayanan}`;
    const fileName = sanitizeFileName_(fileNameBase);

    const docCopy = renderMultiEntryDocument_(payload.templateFileId, entries, fileName);

    // ── ROUTING OUTPUT FOLDER berdasarkan layanan ──────────────────────
    // Kenaikan Pangkat → FOLDER_KP_OUTPUT (tetap Google Docs, bukan PDF)
    // Pensiun (DPCP/SUPER) → FOLDER_PENSIUN_OUTPUT
    // Kontrak Tendik/Dosen → FOLDER_KONTRAK_ROOT / sub-folder tahun
    const layanan = String(payload.layanan || '').trim();
    let outputFolder;
    if (layanan === 'Pensiun') {
      outputFolder = DriveApp.getFolderById(CONFIG.FOLDER_PENSIUN_OUTPUT);
    } else if (layanan === 'Kontrak Tendik' || layanan === 'Kontrak Dosen') {
      const tahun = String(new Date().getFullYear());
      const rootKontrak = DriveApp.getFolderById(CONFIG.FOLDER_KONTRAK_ROOT);
      const existing = rootKontrak.getFoldersByName(tahun);
      outputFolder = existing.hasNext() ? existing.next() : rootKontrak.createFolder(tahun);
    } else {
      // Default: Kenaikan Pangkat → FOLDER_KP_OUTPUT
      outputFolder = DriveApp.getFolderById(CONFIG.FOLDER_KP_OUTPUT);
    }
    pindahFileKeFolder_(docCopy, outputFolder);
    try { docCopy.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch(e) { Logger.log('Share err: '+e.message); }
    // ───────────────────────────────────────────────────────────────────

    // Tandai progres usulan KP (hanya relevan untuk Kenaikan Pangkat)
    if (!layanan || layanan === 'Kenaikan Pangkat') {
      const nipYangDigenerate = entries.map(e => String(e.employee.nip || '').trim());
      nipYangDigenerate.forEach(nip => tandaiOpsiUsulanKpSelesai_(nip, payload.subLayanan, employeeSession.nip));
    }

    return {
      success: true,
      message: entries.length > 1
        ? `Dokumen berhasil dibuat (${entries.length} data digabung dalam satu file).`
        : 'Dokumen berhasil dibuat.',
      fileId: docCopy.getId(),
      fileName: fileName,
      viewUrl: docCopy.getUrl(),
      jumlahEntri: entries.length
    };
  } catch (err) {
    return { success: false, message: 'Gagal generate dokumen: ' + err.message };
  }
}

/**
 * Terima upload Base64 lampiran usulan kontrak dari Vercel → simpan ke
 * sub-folder tahun di FOLDER_KONTRAK_ROOT → kembalikan URL file Drive.
 * Dipanggil oleh ajukanUsulanKontrak di rpc.js melalui doPost.
 */
function ajukanUsulanKontrakDrive(sessionId, payload) {
  try {
    const tahun = String(payload.tahun || new Date().getFullYear());
    const nip   = String(payload.nip || 'unknown').replace(/[^a-zA-Z0-9]/g, '');
    const nama  = sanitizeFileName_(String(payload.nama || 'pegawai').substring(0, 40));

    // Cari / buat sub-folder tahun di dalam root kontrak
    const rootKontrak = DriveApp.getFolderById(CONFIG.FOLDER_KONTRAK_ROOT);
    const tahunIter = rootKontrak.getFoldersByName(tahun);
    const tahunFolder = tahunIter.hasNext() ? tahunIter.next() : rootKontrak.createFolder(tahun);

    // Cari / buat sub-folder pegawai (NIP - Nama) di dalam tahun
    const pegawaiKey = nip + ' - ' + nama;
    const pegIter = tahunFolder.getFoldersByName(pegawaiKey);
    const pegawaiFolder = pegIter.hasNext() ? pegIter.next() : tahunFolder.createFolder(pegawaiKey);

    // Helper: simpan satu file Base64 ke folder
    function simpanBase64_(base64, namaFile, mimeType) {
      if (!base64) return '';
      try {
        const decoded = Utilities.newBlob(
          Utilities.base64Decode(base64.replace(/^data:[^;]+;base64,/, '')),
          mimeType || 'application/octet-stream',
          namaFile
        );
        const file = pegawaiFolder.createFile(decoded);
        file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
        return file.getUrl();
      } catch (e) {
        Logger.log('Gagal simpan ' + namaFile + ': ' + e.message);
        return '';
      }
    }

    function guessMime_(namaFile) {
      const ext = String(namaFile || '').split('.').pop().toLowerCase();
      const map = { pdf: 'application/pdf', jpg: 'image/jpeg', jpeg: 'image/jpeg',
                    png: 'image/png', docx: 'application/vnd.openxmlformats-officedocument.wordprocessingml.document' };
      return map[ext] || 'application/octet-stream';
    }

    const urls = {};
    const files = [
      { key: 'ktp',           base64: payload.ktpBase64,            nama: payload.ktpNama || 'KTP.pdf' },
      { key: 'kk',            base64: payload.kkBase64,             nama: payload.kkNama || 'KK.pdf' },
      { key: 'pas_foto',      base64: payload.pasFotoBase64,        nama: payload.pasFotoNama || 'PasFoto.jpg' },
      { key: 'ijazah',        base64: payload.ijazahBase64,         nama: payload.ijazahNama || 'Ijazah.pdf' },
      { key: 'surat_pengantar', base64: payload.suratPengantarBase64, nama: payload.suratPengantarNama || 'SuratPengantar.pdf' },
      { key: 'surat_lamaran', base64: payload.suratLamaranBase64,   nama: payload.suratLamaranNama || 'SuratLamaran.pdf' },
      { key: 'sim_ab',        base64: payload.simAbBase64,          nama: payload.simAbNama || 'SIM_AB.pdf' },
      { key: 'str_aktif',     base64: payload.strAktifBase64,       nama: payload.strAktifNama || 'STR_Aktif.pdf' },
      { key: 'keterangan_sehat', base64: payload.ketSehatBase64,    nama: payload.ketSehatNama || 'KeteranganSehat.pdf' }
    ];

    files.forEach(f => {
      if (f.base64) {
        urls[f.key] = simpanBase64_(f.base64, f.nama, guessMime_(f.nama));
      }
    });

    return { success: true, urls: urls, folderUrl: pegawaiFolder.getUrl() };
  } catch (err) {
    return { success: false, message: 'GAS Error: ' + err.message };
  }
}

/**
 * Admin generate kontrak final dari usulan yang SUDAH disetujui
 * (semua lampiran approved). Simpan ke folder tahun di FOLDER_KONTRAK_ROOT.
 */
function generateKontrakFromUsulan(sessionToken, templateFileId, usulanData) {
  const adminSession = requireRole_(sessionToken, ['admin', 'super_admin']);
  try {
    if (!usulanData) return { success: false, message: 'Data usulan tidak tersedia.' };

    let fileId = templateFileId;
    if (!fileId) {
      const templates = getTemplatesForService(sessionToken, usulanData.layanan, usulanData.sub_menu);
      if (!templates.length) return { success: false, message: `Template "${usulanData.sub_menu}" (${usulanData.layanan}) belum terdaftar.` };
      fileId = templates[0].fileId;
    }

    // Ambil data pegawai dari sheet berdasarkan NIP
    const nip = String(usulanData.nip || '').trim();
    let employee = {};
    try { employee = getEmployeeFullData(sessionToken, nip) || {}; } catch (e) { /* best effort */ }

    const formData = Object.assign({}, usulanData.form_data || {}, {
      tahun_kontrak: usulanData.tahun,
      jenis_usulan: usulanData.jenis_usulan,
      evaluasi_kinerja: usulanData.evaluasi_kinerja,
      layanan: usulanData.layanan,
      sub_menu: usulanData.sub_menu
    });
    const dataContext = buildDataContext_(employee, formData, usulanData.sub_menu);

    const namaFile = sanitizeFileName_(String(nip + '-' + (employee.nama_lengkap || usulanData.nama || 'kontrak') + '-' + (usulanData.tahun || '')));
    const docCopy = renderTemplateCopy_(fileId, dataContext, namaFile);

    // Simpan ke folder tahun di root kontrak
    const tahun = String(usulanData.tahun || new Date().getFullYear());
    const rootKontrak = DriveApp.getFolderById(CONFIG.FOLDER_KONTRAK_ROOT);
    const tahunIter = rootKontrak.getFoldersByName(tahun);
    const tahunFolder = tahunIter.hasNext() ? tahunIter.next() : rootKontrak.createFolder(tahun);
    pindahFileKeFolder_(docCopy, tahunFolder);
    try { docCopy.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch(e) { Logger.log('Share err: '+e.message); }

    return {
      success: true,
      fileId: docCopy.getId(),
      fileName: namaFile,
      viewUrl: docCopy.getUrl(),
      message: 'Perjanjian Kontrak berhasil dibuat.'
    };
  } catch (err) {
    return { success: false, message: 'Gagal generate kontrak: ' + err.message };
  }
}

/**
 * Menerima base64 file DOCX, mengkonversinya ke PDF via Google Drive,
 * dan mengembalikan link view/download PDF-nya.
 */
function convertDocxToPdf(sessionToken, docxBase64, fileName) {
  requireSession_(sessionToken);
  try {
    const bytes = Utilities.base64Decode(docxBase64);
    const blob = Utilities.newBlob(bytes, 'application/vnd.openxmlformats-officedocument.wordprocessingml.document', fileName);
    
    // Simpan file docx sementara
    const file = DriveApp.createFile(blob);
    
    // Konversi ke PDF
    const pdfBlob = file.getAs('application/pdf');
    pdfBlob.setName(fileName.replace(/\.docx$/i, '') + '.pdf');
    
    // Simpan PDF ke folder root kontrak (atau output default)
    const outputFolder = DriveApp.getFolderById(CONFIG.FOLDER_KONTRAK_ROOT);
    const pdfFile = outputFolder.createFile(pdfBlob);
    
    // Hapus file docx sementara
    file.setTrashed(true);
    
    // Set sharing agar siapa saja yang memiliki link dapat melihat
    try {
      pdfFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    } catch(shareErr) {
      Logger.log('Gagal set sharing PDF: ' + shareErr.message);
    }
    
    return {
      success: true,
      pdfUrl: pdfFile.getUrl(),
      fileName: pdfFile.getName(),
      fileId: pdfFile.getId()
    };
  } catch (err) {
    return { success: false, message: 'Gagal konversi ke PDF: ' + err.message };
  }
}

/**
 * ENTRY POINT WEB APP: Menerima request rpc dari Vercel
 * untuk meng-generate / preview Google Docs.
 */
function doPost(e) {
  try {
    const postData = JSON.parse(e.postData.contents);
    const method = postData.method;
    const params = postData.params || [];

    // ── Injeksi session dari Vercel ──────────────────────────────────────
    // Vercel mengganti JWT panjang (>250 char) dengan UUID pendek agar aman
    // sebagai key CacheService. Data user yang sudah diverifikasi dikirim
    // via field "remoteSession" dan di-inject ke sini sebelum fungsi dipakai.
    if (postData.remoteSession && postData.remoteSession.id && postData.remoteSession.data) {
      try {
        const cache = CacheService.getScriptCache();
        const key   = 'session_' + postData.remoteSession.id;
        const value = JSON.stringify(postData.remoteSession.data);
        cache.put(key, value, 300); // 5 menit, cukup untuk durasi 1 request
      } catch (cacheErr) {
        // Jangan hentikan eksekusi hanya karena CacheService gagal;
        // fungsi yang butuh sesi akan melempar error sendiri jika diperlukan.
        Logger.log('Gagal inject remoteSession ke cache: ' + cacheErr.message);
      }
    }
    // ────────────────────────────────────────────────────────────────────

    const fn = this[method];
    if (typeof fn !== 'function') {
      return ContentService.createTextOutput(JSON.stringify({
        success: false,
        message: 'Fungsi "' + method + '" tidak ditemukan di Apps Script.'
      })).setMimeType(ContentService.MimeType.JSON);
    }
    
    const result = fn.apply(null, params);
    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({
      success: false,
      message: 'Error internal Apps Script: ' + err.message
    })).setMimeType(ContentService.MimeType.JSON);
  }
}
