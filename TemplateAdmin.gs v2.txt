/**
 * =================================================================
 * TEMPLATEADMIN.GS
 * Modul "Template" pada menu utama: registrasi template dokumen
 * Word/Google Docs dari link Drive, disimpan pada sheet "Templates"
 * (dibuat otomatis bila belum ada) di spreadsheet yang sama dengan
 * "Data Utama".
 * =================================================================
 */

const TEMPLATE_HEADERS = ['ID', 'Judul', 'FileId', 'Layanan', 'SubMenu', 'DibuatPada'];

/** Pastikan sheet "Templates" ada, buat beserta header bila belum. */
function ensureTemplatesSheet_() {
  const ss = getSpreadsheet_();
  let sheet = ss.getSheetByName(CONFIG.SHEET_TEMPLATES);
  if (!sheet) {
    sheet = ss.insertSheet(CONFIG.SHEET_TEMPLATES);
    sheet.appendRow(TEMPLATE_HEADERS);
    sheet.setFrozenRows(1);
  }
  return sheet;
}

/**
 * Tambah template baru. Input `driveLink` boleh berupa URL Drive/Docs
 * lengkap ATAU file ID mentah.
 */
function addTemplate(sessionToken, data) {
  requireRole_(sessionToken, ['admin', 'super_admin']);
  try {
    const { driveLink, judul, layanan, subMenu } = data;
    const fileId = extractDriveFileId_(driveLink);
    if (!fileId) return { success: false, message: 'Link Google Drive tidak valid.' };

    // Verifikasi file benar-benar bisa diakses
    let file;
    try {
      file = DriveApp.getFileById(fileId);
    } catch (e) {
      return { success: false, message: 'File tidak ditemukan atau tidak ada akses. Pastikan berbagi sudah benar.' };
    }

    if (!judul || !layanan || !subMenu) {
      return { success: false, message: 'Judul, Layanan, dan Sub-menu wajib diisi.' };
    }

    const sheet = ensureTemplatesSheet_();
    const id = Utilities.getUuid();
    sheet.appendRow([id, judul, fileId, layanan, subMenu, new Date()]);

    return { success: true, message: `Template "${judul}" berhasil ditambahkan.`, id: id };
  } catch (err) {
    return { success: false, message: 'Gagal menambah template: ' + err.message };
  }
}

/** Daftar seluruh template terdaftar (untuk halaman manajemen Template). */
function getAllTemplates(sessionToken) {
  requireSession_(sessionToken);
  const sheet = ensureTemplatesSheet_();
  const values = sheet.getDataRange().getValues();
  if (values.length < 2) return [];

  const rows = [];
  for (let i = 1; i < values.length; i++) {
    const r = values[i];
    if (!r[0]) continue;
    rows.push({
      id: r[0], judul: r[1], fileId: r[2], layanan: r[3], subMenu: r[4],
      dibuatPada: r[5] instanceof Date ? r[5].toISOString() : String(r[5] || '')
    });
  }
  return rows;
}

/**
 * Ambil template yang cocok untuk kombinasi Layanan + SubMenu tertentu
 * (dipakai modul Kenaikan Pangkat untuk menentukan template mana yang
 * dipakai saat tombol Preview/Generate ditekan).
 */
function getTemplatesForService(sessionToken, layanan, subMenu) {
  requireSession_(sessionToken);
  return getAllTemplates(sessionToken).filter(t => t.layanan === layanan && t.subMenu === subMenu);
}

/** Hapus template dari registry (tidak menghapus file Drive aslinya). */
function deleteTemplate(sessionToken, templateId) {
  requireRole_(sessionToken, ['admin', 'super_admin']);
  const sheet = ensureTemplatesSheet_();
  const values = sheet.getDataRange().getValues();
  for (let i = 1; i < values.length; i++) {
    if (values[i][0] === templateId) {
      sheet.deleteRow(i + 1);
      return { success: true, message: 'Template dihapus.' };
    }
  }
  return { success: false, message: 'Template tidak ditemukan.' };
}

/** Daftar Layanan & SubMenu yang tersedia (untuk dropdown form Template). */
function getLayananOptions() {
  return CONFIG.LAYANAN_LIST;
}
